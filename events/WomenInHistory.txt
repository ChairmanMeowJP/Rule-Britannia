# Queen Elizabeth I (1533-1603)
country_event = {
	id = women_history_events.1
	title = women_history_events.1.t
	desc = women_history_events.1.d
	picture = wihgfx_FEMALE_RULER_eventPicture
	
	trigger = {
		is_year = 1533
		NOT = {
			is_year = 1603
		}
		OR = {
			tag = ENG
			tag = GBR
		}
		is_female = no
		has_regency = no
		has_heir = no
		is_lesser_in_union = no
		government = monarchy
		any_owned_province = {
			OR = {
				religion = protestant
				religion = anglican
			}
			is_in_capital_area = yes
		}
		NOT = {
			has_global_flag = wih_elizabeth_eng_flag
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = wih_elizabeth_eng_flag
		}
	}
	
	option = {
		name = women_history_events.1.a
		define_ruler = {
			name = "Elizabeth"
			dynasty = "Tudor"
			culture = english
			fixed = yes
			adm = 6
			dip = 6
			mil = 5
			female = yes
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = inspiring_leader_personality }
		}
	}
	option = {
		name = women_history_events.1.b
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_meritocracy_effect = yes
		define_advisor = {
			culture = english
			female = yes
			type = statesman #spymaster
			skill = 2
			discount = yes
			name = "Elizabeth Tudor"
		}
	}
}

# Gráinne Ní Mháille
country_event = {
	id = women_history_events.22
	title = women_history_events.22.t
	desc = women_history_events.22.d
	picture = wihgfx_FEMALES_eventPicture

	trigger = {
		OR = {
			AND = {
				has_reform = pirate_republic_reform
				has_global_flag = grace_omalley_flag
			}
			AND = {
				owns = 375
				owns = 376
				is_year = 1530
				NOT = {
					is_year = 1605
				}
			}
		}
		NOT = {
			has_global_flag = wih_grainnenimhaille_flag
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = wih_grainnenimhaille_flag
		}
	}
	
	option = {
		name = women_history_events.22.a
		trigger = {
			NOT = { has_global_flag  = grace_omalley_flag }
		}
		add_meritocracy_effect = yes
		define_advisor = {
			female = yes
			type = commandant
			culture = irish
			skill = 2
			name = "Grainne Ni Mhaille"
			discount = yes
		}
	}
	option = {
		name = women_history_events.22.b
		trigger = {
			NOT = { has_global_flag  = grace_omalley_flag }
		}
		define_admiral = {
			name = "Grainne Ni Mhaille"
			shock = 4
			fire = 4
			manuever = 3
			female = yes
		}
	}
	option = {
		name = women_history_events.22.c
		trigger = {
			NOT = { has_global_flag  = grace_omalley_flag }
		}
		add_prestige = 10
	}
	option = {
		name = women_history_events.22.da
		trigger = {
			has_global_flag  = grace_omalley_flag
		}
		define_ruler = {
			name = "Grainne Ni Mhaille"
			adm = 3
			dip = 4
			mil = 5
			fixed = yes
			culture = irish
		}
		if = {
			limit = {
				has_flagship = no
				capital_scope = {
					has_port = yes
				}
			}
			capital_scope = {
				create_flagship = {	
					name = "White Seahorse"
					type = heavy_ship
				}
			}
		}
		else_if = {
			limit = {
				capital_scope = {
					has_port = yes
				}
			}
			capital_scope = {
				create_named_ship = {
					name = "White Seahorse"
					type = heavy_ship
				}
			}
		}
	}
}
